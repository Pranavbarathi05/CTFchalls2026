from cryptography.hazmat.primitives.ciphers import Cipher, algorithms, modes
from cryptography.hazmat.primitives import padding
from cryptography.hazmat.backends import default_backend
import binascii
import os

def run_aes_example():
    # 1. Setup
    flag = "DSCCTF{CTF2026}".encode('utf-8')
    backend = default_backend()
    
    # Key and IV converted from Hex to Bytes
    key = binascii.unhexlify("f7823e42861c841e21713589b9d31f0b")
    iv = binascii.unhexlify("1a5d6290c4b28f9366e741ad2983c501")

    print(f"Original Flag: {flag.decode()}")

    # 2. ENCRYPTION with PKCS7 Padding
    # AES block size is 128 bits (16 bytes)
    padder = padding.PKCS7(128).padder()
    padded_data = padder.update(flag) + padder.finalize()
    
    encryptor = Cipher(algorithms.AES(key), modes.CBC(iv), backend=backend).encryptor()
    ciphertext = encryptor.update(padded_data) + encryptor.finalize()
    
    print(f"Ciphertext (Hex): {binascii.hexlify(ciphertext).decode()}")

    # 3. DECRYPTION
    decryptor = Cipher(algorithms.AES(key), modes.CBC(iv), backend=backend).decryptor()
    decrypted_padded_data = decryptor.update(ciphertext) + decryptor.finalize()
    
    # 4. REMOVE PADDING
    unpadder = padding.PKCS7(128).unpadder()
    try:
        final_plaintext = unpadder.update(decrypted_padded_data) + unpadder.finalize()
        print(f"Decrypted Flag: {final_plaintext.decode('utf-8')}")
    except ValueError as e:
        print(f"Padding Error: {e}")

if __name__ == "__main__":
    run_aes_example()
